def histogram(dataset):
    h = sorted(dataset)                             #sorted
    fit = stats.norm.pdf(h, np.mean(h), np.std(h))  #this is a fitting indeed
    pl.plot(h,fit,'-o')                             #plot data point
    pl.hist(h,normed=True)                          #histogram of your data
    return pl

def get_data(column, dataset, titl, state='State', tex='text',):    
    data = [ dict(
            type='choropleth',
            colorscale = scl,
            autocolorscale = False,
            locations = dataset[state],
            z = dataset[column].astype(int),
            locationmode = 'USA-states',
            text = dataset[tex],
            marker = dict(
                line = dict (
                    color = 'rgb(255,255,255)',
                    width = 2
                ) ),
            colorbar = dict(
                title = titl)
            ) ]
    return data
def get_layout():
    layout = dict(
            title = '2001 US SAT Score',
            geo = dict(
                scope='usa',
                projection=dict( type='albers usa' ),
                showlakes = True,
                lakecolor = 'rgb(255, 255, 255)'),
                 )
    return layout
